### Row Level Security (RLS)

```sql
-- Users: Can only read/update their own data
CREATE POLICY "users_select" ON users
  FOR SELECT USING (auth.uid() = id);

CREATE POLICY "users_update" ON users
  FOR UPDATE USING (auth.uid() = id);

-- Events: Public read, owner write
CREATE POLICY "events_select_public" ON events
  FOR SELECT USING (visibility = 'public');

CREATE POLICY "events_select_restricted" ON events
  FOR SELECT USING (
    visibility = 'restricted' 
    AND (
      created_by_user_id = auth.uid()
      OR EXISTS (
        SELECT 1 FROM event_user_access 
        WHERE event_id = events.id 
        AND user_id = auth.uid()
      )
    )
  );

CREATE POLICY "events_modify" ON events
  FOR ALL USING (created_by_user_id = auth.uid());

-- Clubs: Members can read, owner can write
CREATE POLICY "clubs_select" ON clubs
  FOR SELECT USING (
    EXISTS (
      SELECT 1 FROM club_members
      WHERE club_id = clubs.id
      AND user_id = auth.uid()
    )
  );

CREATE POLICY "clubs_modify" ON clubs
  FOR ALL USING (created_by = auth.uid());
```
