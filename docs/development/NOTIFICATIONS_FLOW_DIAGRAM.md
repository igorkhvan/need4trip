```sql
-- Current queue status
SELECT 
  status,
  COUNT(*) as count,
  MIN(scheduled_for) as oldest,
  MAX(scheduled_for) as newest
FROM notification_queue
GROUP BY status;

-- Failed notifications in last hour
SELECT 
  trigger_type,
  COUNT(*) as failures,
  STRING_AGG(DISTINCT last_error, '; ') as errors
FROM notification_queue
WHERE status = 'failed' 
  AND created_at >= NOW() - INTERVAL '1 hour'
GROUP BY trigger_type;

-- Success rate by trigger (last 24h)
SELECT 
  trigger_type,
  COUNT(CASE WHEN status = 'sent' THEN 1 END) as sent,
  COUNT(CASE WHEN status = 'failed' THEN 1 END) as failed,
  ROUND(
    100.0 * COUNT(CASE WHEN status = 'sent' THEN 1 END) / COUNT(*),
    2
  ) as success_rate
FROM notification_logs
WHERE sent_at >= NOW() - INTERVAL '24 hours'
GROUP BY trigger_type
ORDER BY success_rate ASC;

-- User engagement (disabled triggers)
SELECT 
  ROUND(100.0 * COUNT(CASE WHEN NOT notify_on_datetime_change THEN 1 END) / COUNT(*), 2) as datetime_disabled_pct,
  ROUND(100.0 * COUNT(CASE WHEN NOT notify_on_location_change THEN 1 END) / COUNT(*), 2) as location_disabled_pct,
  ROUND(100.0 * COUNT(CASE WHEN NOT notify_on_rules_change THEN 1 END) / COUNT(*), 2) as rules_disabled_pct
```
